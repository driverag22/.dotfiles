     62 # Confirmation CMD
     63 confirm_cmd() {
     64 	rofi -theme-str 'window {location: center; anchor: center; fullscreen: false; width: 350px;}' \
     65 		-theme-str 'mainbox {orientation: vertical; children: [ "message", "listview" ];}' \
     66 		-theme-str 'listview {columns: 2; lines: 1;}' \
     67 		-theme-str 'element-text {horizontal-align: 0.5;}' \
     68 		-theme-str 'textbox {horizontal-align: 0.5;}' \
     69 		-dmenu \
     70 		-p 'Confirmation' \
     71 		-mesg 'Are you Sure?' \
     72 		-theme ${theme}

     75 # Ask for confirmation

     82 	selected="$(confirm_exit)"
