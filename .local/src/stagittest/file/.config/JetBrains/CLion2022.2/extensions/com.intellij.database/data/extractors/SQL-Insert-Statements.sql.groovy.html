      2  * Available context bindings:
      3  *   COLUMNS     List<DataColumn>
      4  *   ROWS        Iterable<DataRow>

      9  *

     11  *   DataRow     { rowNumber(); first(); last(); data(): List<Object>; value(column): Object }

     15 SEP = ", "
     16 QUOTE     = "\'"
     17 STRING_PREFIX = DIALECT.getDbms().isMicrosoft() ? "N" : ""
     18 NEWLINE   = System.getProperty("line.separator")
     19 
     20 KEYWORDS_LOWERCASE = com.intellij.database.util.DbSqlUtil.areKeywordsLowerCase(PROJECT)
     21 KW_INSERT_INTO = KEYWORDS_LOWERCASE ? "insert into " : "INSERT INTO "
     22 KW_VALUES = KEYWORDS_LOWERCASE ? ") values (" : ") VALUES ("
     23 KW_NULL = KEYWORDS_LOWERCASE ? "null" : "NULL"

     25 def record(columns, dataRow) {
     26     OUT.append(KW_INSERT_INTO)
     27     if (TABLE == null) OUT.append("MY_TABLE")
     28     else OUT.append(TABLE.getParent().getName()).append(".").append(TABLE.getName())
     29     OUT.append(" (")
     30 
     31     columns.eachWithIndex { column, idx ->
     32         OUT.append(column.name()).append(idx != columns.size() - 1 ? SEP : "")
     33     }
     34 
     35     OUT.append(KW_VALUES)
     36     columns.eachWithIndex { column, idx ->
     37         def value = dataRow.value(column)
     38         def stringValue = value == null ? KW_NULL : FORMATTER.formatValue(value, column)
     39         def isStringLiteral = value != null && FORMATTER.isStringLiteral(value, column)
     40         if (isStringLiteral && DIALECT.getDbms().isMysql()) stringValue = stringValue.replace("\\", "\\\\")
     41         OUT.append(isStringLiteral ? (STRING_PREFIX + QUOTE) : "")
     42           .append(isStringLiteral ? stringValue.replace(QUOTE, QUOTE + QUOTE) : stringValue)
     43           .append(isStringLiteral ? QUOTE : "")
     44           .append(idx != columns.size() - 1 ? SEP : "")
     45     }
     46     OUT.append(");").append(NEWLINE)
     47 }

     49 ROWS.each { row -> record(COLUMNS, row) }
