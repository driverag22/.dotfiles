     28 static def splitByLines(values, size) {
     29   def splitValues = new ArrayList<>()
     30   def maxLines = 0
     31   for (int i = 0; i < size; i++) {
     32     def splitValue = StringUtil.splitByLines(values(i))
     33     splitValues.add(splitValue)
     34     maxLines = Math.max(maxLines, splitValue.size())
     35   }
     36 
     37   def byLines = new ArrayList<>(maxLines)
     38   for (int i = 0; i < maxLines; i++) {
     39     def lineValues = new ArrayList<>()
     40     byLines.add(lineValues)
     41     for (int j = 0; j < splitValues.size(); j++) {
     42       def splitValue = splitValues[j]
     43       lineValues.add(splitValue.size() <= i ? null : splitValue[i])
     44     }
     45   }
     46   return byLines
     47 }

     49 def printRow(values, size, width = { COLUMN_WIDTH }, padding = SPACE, separator = { PIPE }) {
     50   def byLines = splitByLines(values, size)
     51   byLines.each { line ->
     52     def lineSize = line.size()
     53     if (lineSize > 0) OUT.append(separator(-1))
     54     for (int i = 0; i < lineSize; i++) {
     55       def value = line[i] == null ? "" : line.get(i)
     56       def curWidth = width(i)
     57       OUT.append(value.padRight(curWidth, padding))
     58       OUT.append(separator(i))
     59     }
     60     OUT.append(NEWLINE)
     61   }
     62 }

     64 def printRows() {
     65   def colNames = COLUMNS.collect { it.name() }
     66   def calcWidth = COLUMN_WIDTH == WIDTH_BASED_ON_CONTENT
     67   def rows
     68   def width
     69   def rowFormatter
     70   if (calcWidth) {
     71     rows = new ArrayList<>()
     72     def widths = new int[COLUMNS.size()]
     73     COLUMNS.eachWithIndex { column, idx -> widths[idx] = column.name().length() }
     74     ROWS.each { row ->
     75       def rowValues = COLUMNS.withIndex().collect { col, idx ->
     76         def value = FORMATTER.format(row, col)
     77         widths[idx] = Math.max(widths[idx], value.length())
     78         value
     79       }
     80       rows.add(rowValues)
     81     }
     82     width = { widths[it] }
     83     rowFormatter = { it }
     84   }
     85   else {
     86     rows = ROWS
     87     width = { COLUMN_WIDTH }
     88     rowFormatter = { COLUMNS.collect { col -> FORMATTER.format(it, col) } }
     89   }
     90 
     91   printRow({""}, COLUMNS.size(), { width(it) }, MINUS) { CROSS }
     92   printRow( { colNames[it] }, COLUMNS.size()) { width(it) }
     93 
     94   def first = true
     95   rows.each { row ->
     96     def rowValues = rowFormatter(row)
     97     if (first || ROW_SEPARATORS) printRow({""}, COLUMNS.size(), { width(it) }, MINUS) { first ? CROSS : MINUS }
     98     printRow({ rowValues[it] }, rowValues.size()) { width(it) }
     99     first = false
    100   }
    101   printRow({""}, COLUMNS.size(), { width(it) }, MINUS) { CROSS }
    102 }

    104 def printRowsTransposed() {
    105   def calcWidth = COLUMN_WIDTH == WIDTH_BASED_ON_CONTENT
    106   if (calcWidth) {
    107     COLUMN_WIDTHS = new ArrayList<Integer>()
    108     COLUMN_WIDTHS.add(0)
    109   }
    110   def valuesByRow = COLUMNS.collect { col ->
    111     if (calcWidth) COLUMN_WIDTHS.set(0, Math.max(COLUMN_WIDTHS[0], col.name().length()))
    112     new ArrayList<String>([col.name()])
    113   }
    114   def rowCount = 1
    115   ROWS.each { row ->
    116     rowCount++
    117     COLUMNS.eachWithIndex { col, i ->
    118       def formattedValue = FORMATTER.format(row, col)
    119       valuesByRow[i].add(formattedValue)
    120       def widthIdx = rowCount - 1
    121       def length = formattedValue.length()
    122       if (calcWidth) {
    123         if (COLUMN_WIDTHS.size() == widthIdx) COLUMN_WIDTHS.add(length)
    124         COLUMN_WIDTHS.set(widthIdx, Math.max(COLUMN_WIDTHS[widthIdx], length))
    125       }
    126     }
    127   }
    128   valuesByRow.each { row ->
    129     printRow({ "" }, rowCount, { calcWidth ? COLUMN_WIDTHS[it] : COLUMN_WIDTH }, MINUS) {
    130       it <= 0 ? CROSS : it == rowCount - 1 ? CROSS : MINUS
    131     }
    132     printRow({ row[it] }, row.size()) { calcWidth ? COLUMN_WIDTHS[it] : COLUMN_WIDTH }
    133   }
    134   printRow({ "" }, rowCount, { calcWidth ? COLUMN_WIDTHS[it] : COLUMN_WIDTH }, MINUS) {
    135     it <= 0 ? CROSS : it == rowCount - 1 ? CROSS : MINUS
    136   }
    137 }
