     19 INDENT = "  "

     21 def printJSON(level, col, o) {
     22   switch (o) {
     23     case null: OUT.append("null"); break
     24     case Tuple: printJSON(level, o[0], o[1]); break
     25     case Map:
     26       OUT.append("{")
     27       o.entrySet().eachWithIndex { entry, i ->
     28         OUT.append("${i > 0 ? "," : ""}$NEWLINE${INDENT * (level + 1)}")
     29         OUT.append("\"${escapeStr(entry.getKey().toString())}\"")
     30         OUT.append(": ")
     31         printJSON(level + 1, col, entry.getValue())
     32       }
     33       OUT.append("$NEWLINE${INDENT * level}}")
     34       break
     35     case Object[]:
     36     case Iterable:
     37       OUT.append("[")
     38       def plain = true
     39       o.eachWithIndex { item, i ->
     40         plain = item == null || item instanceof Number || item instanceof Boolean || item instanceof String
     41         if (plain) {
     42           OUT.append(i > 0 ? ", " : "")
     43         }
     44         else {
     45           OUT.append("${i > 0 ? "," : ""}$NEWLINE${INDENT * (level + 1)}")
     46         }
     47         printJSON(level + 1, col, item)
     48       }
     49       if (plain) OUT.append("]") else OUT.append("$NEWLINE${INDENT * level}]")
     50       break
     51     case Boolean: OUT.append("$o"); break
     52     default:
     53       def str = FORMATTER.formatValue(o, col)
     54       def typeName = FORMATTER.getTypeName(o, col)
     55       def shouldQuote = FORMATTER.isStringLiteral(o, col) && !(typeName.equalsIgnoreCase("json") || typeName.equalsIgnoreCase("jsonb"))
     56       OUT.append(shouldQuote ? "\"${escapeStr(str)}\"" : str);
     57       break
     58   }
     59 }
