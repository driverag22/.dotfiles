     23 def escapeTag(name) {
     24   name = pattern.matcher(name).replaceAll("_")
     25   return name.isEmpty() || !Character.isLetter(name.charAt(0)) ? "_$name" : name
     26 }
     27 def printRow(level, rowTag, values) {
     28   def prefix = "$NEWLINE${StringUtil.repeat("  ", level)}"
     29   OUT.append("$prefix<$rowTag>")
     30   values.each { name, col, valuesName, value ->
     31     switch (value) {
     32       case Map:
     33         def mapValues = new ArrayList<Tuple>()
     34         value.each { key, v -> mapValues.add(new Tuple(escapeTag(key.toString()), col, key.toString(), v)) }
     35         printRow(level + 1, name, mapValues)
     36         break
     37       case Object[]:
     38       case Iterable:
     39         def listItems = new ArrayList<Tuple>()
     40         def itemName = valuesName != null ? escapeTag(StringUtil.unpluralize(valuesName) ?: "item") : "item"
     41         value.collect { v -> listItems.add(new Tuple(itemName, col, null, v)) }
     42         printRow(level + 1, name, listItems)
     43         break
     44       default:
     45         OUT.append("$prefix  <$name>")
     46         if (value == null) OUT.append("null")
     47         else {
     48           def formattedValue = FORMATTER.formatValue(value, col)
     49           if (isXmlString(formattedValue)) OUT.append(formattedValue)
     50           else OUT.append(StringUtil.escapeXmlEntities(formattedValue))
     51         }
     52         OUT.append("</$name>")
     53     }
     54   }
     55   OUT.append("$prefix</$rowTag>")
     56 }
