     57 def calcFields(table) {
     58   DasUtil.getColumns(table).reduce([]) { fields, col ->
     59     def spec = Case.LOWER.apply(col.getDataType().getSpecification())
     60     def typeStr = typeMapping.find { p, t -> p.matcher(spec).find() }.value
     61     fields += [[
     62                  name : javaName(col.getName(), false),
     63                  type : typeStr,
     64                  annos: ""]]
     65   }
     66 }

     68 def javaName(str, capitalize) {
