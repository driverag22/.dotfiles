      2  * Available context bindings:
      3  *   COLUMNS     List<DataColumn>
      4  *   ROWS        Iterable<DataRow>
      5  *   OUT         { append() }
      6  *   FORMATTER   { format(row, col); formatValue(Object, col); getTypeName(Object, col); isStringLiteral(Object, col); }
      7  *   TRANSPOSED  Boolean
      8  * plus ALL_COLUMNS, TABLE, DIALECT
      9  *
     10  * where:
     11  *   DataRow     { rowNumber(); first(); last(); data(): List<Object>; value(column): Object }
     12  *   DataColumn  { columnNumber(), name() }
     13  */
     14 
     15 import static com.intellij.openapi.util.text.StringUtil.escapeXmlEntities
     16 
     17 NEWLINE = System.getProperty("line.separator")
     18 
     19 def HTML_PATTERN = ~"<.+>"
     20 
     21 def printRow = { values, tag, valueToString ->
     22   OUT.append("$NEWLINE<tr>$NEWLINE")
     23   values.each {
     24     def str = valueToString(it)
     25     def escaped = str ==~ HTML_PATTERN
     26       ? str
     27       : escapeXmlEntities((str as String).replaceAll("\\t|\\b|\\f", "")).replaceAll("\\r|\\n|\\r\\n", "<br/>")
     28     OUT.append("  <$tag>$escaped</$tag>$NEWLINE")
     29   }
     30   OUT.append("</tr>")
     31 }
     32 
     33 OUT.append(
     34 """<!DOCTYPE html>
     35 <html>
     36   <head>
     37     <title></title>
     38     <meta charset="UTF-8">
     39   </head>
     40 <body>
     41 <table border="1" style="border-collapse:collapse">""")
     42 
     43 if (!TRANSPOSED) {
     44   printRow(COLUMNS, "th") { it.name() }
     45   ROWS.each { row -> printRow(COLUMNS, "td") { FORMATTER.format(row, it) } }
     46 }
     47 else {
     48   def values = COLUMNS.collect { new ArrayList<String>( [it.name()] ) }
     49   ROWS.each { row -> COLUMNS.eachWithIndex { col, i -> values[i].add(FORMATTER.format(row, col)) } }
     50   values.each { printRow(it, "td", { it }) }
     51 }
     52 
     53 OUT.append("""
     54 </table>
     55 </body>
     56 </html>
     57 """)
